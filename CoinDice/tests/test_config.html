from CoinDice.app import create_app
from CoinDice.settings import ProdConfig, DevConfig
#

This Tests Our Production Config Tests Our Production Config.

def test_production_config():
# 
    app = create_app(ProdConfig) # This Initiates Our Application Using Our Production Config
    assert app.config['ENV'] == 'prod' # If In Production The Enviroment Must Be 'prod'
    assert app.config['DEBUG'] is False # Its Unsage to be in Debug Mode As We Expose Sessions, Code and Variables
    assert app.config['DEBUG_TB_ENABLED'] is False # This Will Give Users Un-Needed Info Which Is Definitely Unwanted For a Security Orientated Application
    assert app.config['ASSETS_DEBUG'] is False  # As Above
#

Tests Our Development Config.

def test_dev_config():
# 
    app = create_app(DevConfig)
    assert app.config['ENV'] == 'dev' # When Developing We Want to Be Able To Use The Full Range Of Tools
    assert app.config['DEBUG'] is True # When Developing its easier having debug enabled so we can see all the issues
    assert app.config['ASSETS_DEBUG'] is True # As Above
