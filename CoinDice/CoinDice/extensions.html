extensions.py

#

Extensions module. Each extension is initialized in the app factory located in app.py


    

  

  

  

    

      

        #
      

      
Our Password Hashing Module

    

    

      
from flask.ext.bcrypt import Bcrypt
bcrypt = Bcrypt()

    

  

  

  

    

      

        #
      

      
Manages User Login Sessions

    

    

      
from flask.ext.login import LoginManager
login_manager = LoginManager()

    

  

  

  

    

      

        #
      

      
SQLAlchemy for database communication

    

    

      
from flask.ext.sqlalchemy import SQLAlchemy
db = SQLAlchemy()

    

  

  

  

    

      

        #
      

      
Helps to manage our database and deal with updates

    

    

      
from flask.ext.migrate import Migrate
migrate = Migrate()

    

  

  

  

    

      

        #
      

      
Caches Data

    

    

      
from flask.ext.cache import Cache
cache = Cache()

    

  

  

  

    

      

        #
      

      
Debug Toolbar shows us HTTP Headers, Session, Request Vars, Time, Load, Profiling Direct on the site when debug mode is enabled

    

    

      
from flask.ext.debugtoolbar import DebugToolbarExtension
debug_toolbar = DebugToolbarExtension()
