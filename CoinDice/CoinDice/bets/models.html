import datetime as dt
from CoinDice.database import (
    Column,
    db,
    Model,
    ReferenceCol,
    relationship,
    SurrogatePK,
)
#

Our Bets Table

class Bets(SurrogatePK, Model):
    bet_id     = Column(db.Integer, unique=True, nullable=False) # Our Bet ID Number
    user_id    = Column(db.Integer, nullable=False)		 # Our User ID Number
    bet_time   = Column(db.DateTime, default=dt.datetime.utcnow) # DT Bet was placed
    bet_amount = Column(db.Float)				 # Bet Amount
    payout     = Column(db.Float)				 # Total Winnings
    game       = Column(db.String(5))				 # Required Number
    roll       = Column(db.Float)				 # Number Rolled
    profit     = Column(db.Float)				 # Total Profit
    result     = Column(db.String(4))				 # Win/Lose
    s_seed     = Column(db.String(32))				 # Server Seed
    c_seed     = Column(db.String(32))				 # Client Seed
    nonce      = Column(db.String(128))				 # Game Nonce
# 
    def __repr__(self):
        return '<Bet({bet_id})>'.format(bet_id=self.bet_id)
# 
class Seeds(SurrogatePK, Model):
    server_seed = Column(db.String(32), unique=True, nullable=False) # Server Seed
    hashed_seed = Column(db.String(128), nullable=False)	     # Hashed Seed
    start_time  = Column(db.DateTime)				     # DT in use 
    end_time    = Column(db.DateTime)				     # DT finished
