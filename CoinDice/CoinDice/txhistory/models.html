import datetime as dt
from CoinDice.database import (
    Column,
    db,
    Model,
    ReferenceCol,
    relationship,
    SurrogatePK,
)
#

Site Wide Transactions

class Transactions(SurrogatePK, Model):
    tx_id      = Column(db.String(64), unique=True, nullable=False)	# Transaction ID Primary Key
    tx_type    = Column(db.String(8), nullable=False)			# Transaction Type	
    tx_time    = Column(db.DateTime, default=dt.datetime.utcnow)	# Transaction Date/Time
    user_id    = Column(db.Integer)					# User ID 
    amount     = Column(db.Float)					# Amount of Transaction
    w_address  = Column(db.String(35))					# Withdrawal Address if exists
# 
    def __repr__(self):
        return '<TX({tx_id})>'.format(tx_id=self.tx_id)
